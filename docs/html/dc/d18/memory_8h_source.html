235 int memory_get_protection(const void* address)
236 {
237  int result = 0;
238 
239  if (address != NULL) {
240 #ifdef __WINDOWS__
241  MEMORY_BASIC_INFORMATION page_info;
242 
243  if (VirtualQuery(address, &page_info, sizeof(MEMORY_BASIC_INFORMATION)) == sizeof(MEMORY_BASIC_INFORMATION))
244  result = page_info.Protect;
245 #else
246  FILE* maps = fopen("/proc/self/maps", "r");
247 
248  if (maps != NULL) {
249  ulong_t block[2] = { 0, 0 }; /* block of memory addresses */
250  char perms[5]; /* set of permissions */
251 
252  /* parse file lines */
253  #ifdef __X86_ARCH__
254  while (fscanf(maps, "%x-%x %4s %*x %*d:%*d %*d %*s\n", &block[0], &block[1], perms) == 3) {
255  #else
256  while (fscanf(maps, "%lx-%lx %4s %*x %*d:%*d %*d %*s\n", &block[0], &block[1], perms) == 3) {
257  #endif
258  if (block[0] <= (ulong_t)address && block[1] >= (ulong_t)address) {
259  result |= (perms[0] == 'r') ? PROT_READ : PROT_NONE; /* can be readed */
260  result |= (perms[1] == 'w') ? PROT_WRITE : PROT_NONE; /* can be written */
261  result |= (perms[2] == 'x') ? PROT_EXEC : PROT_NONE; /* can be executed */
262 
263  break;
264  }
265  }
266 
267  fclose(maps);
268  }
269 #endif
270  }
271 
272  return result;
273 }
