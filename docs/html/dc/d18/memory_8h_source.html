123 void memory_set_protection(const void* address, const size_t size, const int mode)

147 int memory_get_protection(const void* address)
148 {
149  int result = 0;
150 
151  if (address != NULL) {
152 #ifdef __WINDOWS__
153  MEMORY_BASIC_INFORMATION page_info;
154 
155  if (VirtualQuery(address, &page_info, sizeof(MEMORY_BASIC_INFORMATION)) == sizeof(MEMORY_BASIC_INFORMATION))
156  result = page_info.Protect;
157 #else
158  FILE* maps = fopen("/proc/self/maps", "r");
159 
160  if (maps != NULL) {
161  ulong_t block[2] = { 0, 0 }; /* block of memory addresses */
162  char perms[5]; /* set of permissions */
163 
164  /* parse file lines */
165  #ifdef __X86_ARCH__
166  while (fscanf(maps, "%x-%x %4s %*x %*d:%*d %*d %*s\n", &block[0], &block[1], perms) == 3) {
167  #else
168  while (fscanf(maps, "%lx-%lx %4s %*x %*d:%*d %*d %*s\n", &block[0], &block[1], perms) == 3) {
169  #endif
170  if (block[0] <= (ulong_t)address && block[1] >= (ulong_t)address) {
171  result |= (perms[0] == 'r') ? PROT_READ : PROT_NONE; /* can be readed */
172  result |= (perms[1] == 'w') ? PROT_WRITE : PROT_NONE; /* can be written */
173  result |= (perms[2] == 'x') ? PROT_EXEC : PROT_NONE; /* can be executed */
174 
175  break;
176  }
177  }
178 
179  fclose(maps);
180  }
181 #endif
182  }
183 
184  return result;
185 }

195 void memory_set_raw(void* address, const void* data, const size_t size, const bool vp)
196 {
197  if (address != NULL && data != NULL && size > 0) {
