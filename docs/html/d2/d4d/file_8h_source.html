134 bool file_read(file_t* file, const OFF_T offset, const OFF_T count, const bool change_indicator)
135 {
136  if (file != NULL) {
137  if (file->handle != NULL && (size_t)offset < file->size) {
138  OFF_T old_pos = (!change_indicator) ? FTELL(file->handle) : 0;
139 
140  if (!FSEEK(file->handle, offset, SEEK_SET)) {
141  file->offset = offset;
142 
143  /* maximum or given buffer size without end of line char */
144  file->buffer.size = ((size_t)(offset + count) > file->size) ? file->size - (size_t)offset : (size_t)count;
145 
146  if (data_alloc(&file->buffer)) {
147  if (fread(file->buffer.address, 1, file->buffer.size, file->handle) == file->buffer.size)

153  data_free(&file->buffer);
154  }
155 
156  /* reset file position indicator */
157  if (!change_indicator)
158  FSEEK(file->handle, old_pos, SEEK_SET);

207 bool file_write(file_t* file, const OFF_T offset, const OFF_T count, const bool change_indicator)
208 {
209  if (file != NULL) {
210  if (file->handle != NULL && file->buffer.address != NULL && (size_t)offset < file->size) {
211  OFF_T old_pos = (!change_indicator) ? FTELL(file->handle) : 0;
212 
213  if (!FSEEK(file->handle, offset, SEEK_SET)) {
214  /* prevents reading data outside buffer */
215  size_t size = ((size_t)count > file->buffer.size) ? file->buffer.size : (size_t)count;
216 
217  if (fwrite(file->buffer.address, 1, size, file->handle) == size) {
218  file->size = (size_t)file_find_size(file);
219 
220  return true;
221  }
222  }
223 
224  /* reset file position indicator */
225  if (!change_indicator)
226  FSEEK(file->handle, old_pos, SEEK_SET);
227  }
228  }
229 
230  return false;
