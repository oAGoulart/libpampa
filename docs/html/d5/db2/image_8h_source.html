96 bool image_load(image_t** image, char* path)
97 {
98  bool successful = false;
99 
100  if (path != NULL && image != NULL) {
101  file_t* img = NULL;
102  if (!file_open(&img, path))
103  return successful;
104 
105  *image = (image_t*)calloc(sizeof(image_t), 1);
106  if (*image == NULL) {
107  if (is_png(img)) {
108  if (file_seek(img, 8)) {
109  size_t chunks = 0;
110  bool valid = true;
111 
112  while (file_read(img, 0, 8, true)) {
113  chunks++;
114  uint32_t length = *(uint32_t*)img->buffer.address;
115  ubyte_t* type = img->buffer.address + 4;
116 
117  if (!memcmp("IHDR", type, 4)) {
118  if (chunks != 1 || length != 13 || !file_read(img, 0, length, true)) {
119  valid = false;
120  break;
121  }
122 
123  (*image)->width = (size_t)*(int32_t*)img->buffer.address;
124  (*image)->height = (size_t)*(int32_t*)img->buffer.address + 4;
125  (*image)->depth = (ubyte_t)*(int8_t*)img->buffer.address + 8;
126  (*image)->color = (ubyte_t)*(int8_t*)img->buffer.address + 9;
127  (*image)->compression = (ubyte_t)*(int8_t*)img->buffer.address + 10;
128  (*image)->filter = (ubyte_t)*(int8_t*)img->buffer.address + 11;
129  (*image)->interlace = (ubyte_t)*(int8_t*)img->buffer.address + 12;
130 
131  if ((*image)->compression || (*image)->filter || (*image)->interlace > 1)
132  LOG("Warning: PNG format has invalid values.");
133  }
134 
135  if (!memcmp("IDAT", type, 4)) {
136  if (chunks == 1 || !file_read(img, 0, length, true)) {
137  valid = false;
138  break;
139  }
140 
141  if (!data_append(&img->buffer, &(*image)->buffer))
142  LOG("Warning: Could not properly load image, output may be corrupted.");
143  }
144 
145  if (!memcmp("IEND", type, 4)) {
146  if (chunks == 1)
147  valid = false;
148  break;
149  }
150 
151  (void)file_seek(img, img->offset + 4);
152  }
153  successful = valid;
154  }
155  }
156  }
157 
158  file_close(img);
159  }
160 
161  if (!successful)
162  image_unload(*image);
163  return successful;
164 }
